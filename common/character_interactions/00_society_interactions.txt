recruit_mission_interaction = {
	category = interaction_category_friendly
	common_interaction = yes
	interface_priority = 60
	desc = recruit_mission_interaction_desc
	
	ai_targets = {
		ai_recipients = liege
	}
	ai_targets = {
		ai_recipients = neighboring_rulers
		ai_recipients = peer_vassals
		max = 10
	}
	ai_targets = {
		ai_recipients = guests
	}
	ai_targets = {
		ai_recipients = vassals
		max = 10
	}
	ai_target_quick_trigger = {
		adult = yes
	}
	ai_will_do = {
		always = yes
	}
	ai_frequency = 60
	
	is_shown = {
		NOT = {
			scope:recipient = scope:actor
		}
		scope:actor = {
			has_variable = mission_recruitment
		}
		scope:recipient = {
			NOT = {
				is_in_any_society = yes
			}
		}
	}
	
	can_send = {
		NOT = {
			scope:recipient = scope:actor
		}
		scope:actor = {
			has_variable = mission_recruitment
		}
	}
	
	
	
	is_valid_showing_failures_only = {
		scope:actor = {
			NOT = { has_trait = incapable }
		}
	}
	
	on_accept = {
		scope:actor = {
			change_variable = {
				name = society_currency
				add = 300
			}
			remove_variable = society_mission
			remove_variable = mission_recruitment
		}
		recruit_to_your_society = yes
	}
}

sacrifice_mission_interaction = {
	category = interaction_category_hostile
	common_interaction = yes
	interface_priority = 60
	desc = sacrifice_mission_interaction_desc
	
	ai_targets = {
		ai_recipients = liege
	}
	ai_targets = {
		ai_recipients = neighboring_rulers
		ai_recipients = peer_vassals
		max = 10
	}
	ai_targets = {
		ai_recipients = guests
	}
	ai_targets = {
		ai_recipients = vassals
		max = 10
	}
	ai_target_quick_trigger = {
		adult = yes
	}
	ai_will_do = {
		base = 10
		modifier = {
			add = 100
			has_trait = sadistic
		}
	}
	ai_frequency = 60
	
	is_shown = {
		NOT = {
			scope:recipient = scope:actor
		}
		scope:actor = {
			has_variable = mission_sacrifice
		}
	}
	
	can_send = {
		NOT = {
			scope:recipient = scope:actor
		}
		scope:recipient = { is_pool_guest_of = scope:actor }
		scope:actor = {
			has_variable = mission_sacrifice
		}
	}
	
	
	
	is_valid_showing_failures_only = {
		scope:actor = {
			NOT = { has_trait = incapable }
		}
	}
	
	on_accept = {
		scope:actor = {
			change_variable = {
				name = society_currency
				add = 300
			}
			remove_variable = society_mission
			remove_variable = mission_sacrifice
			add_secret = {
				type = secret_murder
				target = scope:recipient
			}
		}
		scope:recipient = {
			death = death_disappearance
		}
	}
}

poision_ability_interaction = {
	category = interaction_category_hostile
	common_interaction = yes
	interface_priority = 60
	desc = poision_ability_interaction_desc
	
	ai_targets = {
		ai_recipients = liege
	}
	ai_targets = {
		ai_recipients = neighboring_rulers
		ai_recipients = peer_vassals
		max = 10
	}
	ai_targets = {
		ai_recipients = guests
	}
	ai_targets = {
		ai_recipients = vassals
		max = 10
	}
	ai_will_do = {
		base = 0
		modifier = {
			add = 100
			has_trait = sadistic
		}
	}
	ai_frequency = 60
	
	is_shown = {
		NOT = {
			scope:recipient = scope:actor
		}
		scope:actor = {
			ability_poison_is_society_valid = { SOCIETY = society_demonic }
		}
	}
	
	can_send = {
		NOT = {
			scope:recipient = scope:actor
		}
		scope:actor = {
			has_variable = society_demonic_member
		}
	}
	
	
	
	is_valid_showing_failures_only = {
		scope:actor = {
			NOT = { has_trait = incapable }
		}
	}
	
	on_accept = {
		scope:actor = {
			change_variable = {
				name = society_currency
				subtract = 300
			}
		}
		scope:recipient = {
			add_trait = bubonic_plague
		}
	}
}
